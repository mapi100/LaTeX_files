\section*{Solutions to the Set 2}
\subsection*{Problem 1}
\begin{enumerate}
    \item[(a)] 
    \begin{align*}
    y^2 + 2y - 12x + 61 &= 0 \\
    y^2 + 2y &= 12x - 61 \\
    (y^2 + 2y + 1) &= 12x - 61 + 1 \\
    (y + 1)^2 &= 12x - 60 \\
    (y + 1)^2 &= 12(x - 5) \\
    \text{Standard form: } (y - k)^2 &= 4p(x - h) \\
    \text{Vertex: } (5, -1) \\
    4p = 12 &\implies p = 3 \\
    \text{Focus: } (5 + 3, -1) &= (8, -1) \\
    \text{Directrix: } x &= 5 - 3 = 2
    \end{align*}

    \item[(b)] 
    \begin{align*}
    -2x^2 + 8x - 4y - 24 &= 0 \\
    -2x^2 + 8x &= 4y + 24 \\
    -2(x^2 - 4x) &= 4y + 24 \\
    -2(x^2 - 4x + 4) &= 4y + 24 - 8 \\
    -2(x - 2)^2 &= 4y + 16 \\
    (x - 2)^2 &= -2y - 8 \\
    (x - 2)^2 &= -2(y + 4) \\
    \text{Standard form: } (x - h)^2 &= 4p(y - k) \\
    \text{Vertex: } (2, -4) \\
    4p = -2 &\implies p = -0.5 \\
    \text{Focus: } (2, -4 - 0.5) &= (2, -4.5) \\
    \text{Directrix: } y &= -4 + 0.5 = -3.5
    \end{align*}
\end{enumerate}
