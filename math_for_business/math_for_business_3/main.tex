\section*{Solutions to the Set 1}
\subsection*{Problem 1}
\subsubsection*{For \( g(x) = \dfrac{3}{5}x - 7 \)}

\begin{itemize}
    \item \text{\(x\)-intercept:}
    \[
    0 = \dfrac{3}{5}x - 7 \Rightarrow \dfrac{3}{5}x = 7 \Rightarrow x = \dfrac{35}{3}
    \]
    
    \item \text{\(y\)-intercept:}
    \[
    g(0) = -7
    \]
    
    \item \text{Slope:} \( m = \dfrac{3}{5} \)
\end{itemize}

\subsubsection*{For \( f(x) = 3x + 8 \)}

\begin{itemize}
    \item \text{\(x\)-intercept:}
    \[
    0 = 3x + 8 \Rightarrow x = -\dfrac{8}{3}
    \]
    
    \item \text{\(y\)-intercept:}
    \[
    f(0) = 8
    \]
    
    \item \text{Slope:} \( m = 3 \)
\end{itemize}

\section*{Solutions to the Set 2}
\subsection*{Problem 1}
\begin{itemize}
    \item (a) \(m=-\dfrac{2}{3}\)
    \item (b) \begin{enumerate}
    \item \text{Find perpendicular slope} ($m_2$):
    \begin{align*}
        m_1 \cdot m_2 &= -1 \\
        -\tfrac{2}{3} \cdot m_2 &= -1 \\
        m_2 &= \tfrac{3}{2}
    \end{align*}
    
    \item \text{Use point-slope form} with $(2,-3)$:
    \begin{align*}
        y - y_1 &= m_2(x - x_1) \\
        y - (-3) &= \tfrac{3}{2}(x - 2) \\
        y + 3 &= \tfrac{3}{2}x - 3
    \end{align*}
    
    \item \text{Convert to general form}:
    \begin{align*}
        2(y + 3) &= 3x - 6 \quad \text{(Multiply by 2)} \\
        3x - 2y - 12 &= 0
    \end{align*}
\end{enumerate}
\end{itemize}
