\section*{Problem Set 1\\Difficulty level: Normal}
\subsection*{Problem 1}
\textbf{Maximize:} \( P = 5x + 7y \) \\
\textbf{Subject to:}
\[
\left\{
\begin{array}{lcr}
2x + 3y & \leq & 45 \\
x - 3y & \geq & 2 \\
x, y & \geq & 0
\end{array}
\right.
\]

\subsection*{Problem 2}
\textbf{Maximize:} \( Z = 4x - 10y \) \\
\textbf{Subject to:}
\[
\left\{
\begin{array}{lcr}
x - 4y & \geq & 4 \\
2x - y & \leq & 2 \\
x, y & \geq & 0
\end{array}
\right.
\]

\subsection*{Problem 3}
\textbf{Minimize:} \( Z = 20x + 30y \) \\
\textbf{Subject to:}
\[
\left\{
\begin{array}{lcr}
2x + y & \leq & 10 \\
3x + 4y & \leq & 24 \\
8x + 7y & \geq & 56 \\
x, y & \geq & 0
\end{array}
\right.
\]

\subsection*{Problem 4}
\textbf{Minimize:} \( C = 2x + 2y \) \\
\textbf{Subject to:}
\[
\left\{
\begin{array}{lcr}
x + 2y & \geq & 80 \\
3x + 2y & \geq & 160 \\
5x + 2y & \geq & 200 \\
x, y & \geq & 0
\end{array}
\right.
\]
