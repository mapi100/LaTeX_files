\section*{Problem 1}

\begin{itemize}
    \item[a.] 200 grams, 121 grams
    \item[b.] 400 grams, 242 grams
    \item[c.] 600 grams, 363 grams
    \item[d.] 400 grams, 355 grams
\end{itemize}
